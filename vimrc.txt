"Copy all matshes. Test "v, select text, Ctrl-C, :\%Vpattern, enter, :let @a='',
":CopyMatches a, paste
function! CopyMatches(reg)
  let hits = []
  %s//\=len(add(hits, submatch(0))) ? submatch(0) : ''/gne
  let reg = empty(a:reg) ? '+' : a:reg
  execute 'let @'.reg.' = join(hits, "\n") . "\n"'
endfunction
command! -register CopyMatches call CopyMatches(<q-reg>)

function! Crc()
let value=strftime("%y%m%d%H%M%S")
"echo value
let a = system("echo -n " . value . " | cksum | cut -d' ' -f1 | tr -d '\n' ")
let code = printf("%x", str2nr(a,10))
return code 
endfunction
